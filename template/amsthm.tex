% Standalone theorem with proof template.
\documentclass[preview,margin=1pt]{standalone}
\usepackage{amsmath,amssymb}
\input{AniLaTeX-preamble}
\usepackage{amsthm}
\newtheorem*{theorem}{\input{AniLaTeX-theorem={Theorem}}}
\renewcommand*{\proofname}{\input{AniLaTeX-proof={Proof}}}
\begin{AniLaTeX-boldname}
% Reference: https://tex.stackexchange.com/a/43971
\makeatletter
\def\th@plain{%
  \thm@notefont{}% same as heading font
  \itshape% body font
}
\def\th@definition{%
  \thm@notefont{}% same as heading font
  \normalfont% body font
}
\makeatother
\end{AniLaTeX-boldname}
\begin{document}
\begin{theorem}[\input{AniLaTeX-name}]
\input{AniLaTeX-body:1={Content of theorem}}
\end{theorem}
\begin{AniLaTeX-body:2}
\begin{proof}\renewcommand{\qedsymbol}{\input{AniLaTeX-symbol={$\square$}}}
\input{AniLaTeX-body:2={Proof or remarks}}
\end{proof}
\end{AniLaTeX-body:2}
\end{document}
